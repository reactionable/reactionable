### ðŸ§¾ Usage

#### Setup GraphQL Client

##### 1. Wrap the root component of you application in Reactionable App (E.g: `index.tsx`):

###### In a React App

```tsx
import { App } from "@reactionable/core/lib/app/App";
import { GraphqlProvider } from "@reactionable/graphql/lib/Client";
import { ReactElement } from "react";

const graphqlUri = "http://localhost/graphql";

const MyApp = (): ReactElement => {
  return (
    <GraphqlProvider uri={graphqlUri}>
      <App />
    </GraphqlProvider>
  );
};
export default MyApp;
```

###### In a NextJs App

```tsx
import { App } from "@reactionable/core/lib/app/App";
import { GraphqlProvider } from "@reactionable/graphql/lib/Client";
import { ReactElement } from "react";

const graphqlUri = "http://localhost/graphql";

const MyApp = ({ Component, pageProps }): ReactElement => {
  return (
    <GraphqlProvider uri={graphqlUri}>
      <App>
        <Component {...pageProps} />
      </App>
    </GraphqlProvider>
  );
};
export default MyApp;
```
