#### Setup App

###### In a React App

```tsx
import { App } from "@reactionable/core/lib/app/App";
import { useUIProviderProps } from "@reactionable/ui-bootstrap";
import { ReactElement } from "react";

const MyApp = (): ReactElement => {
  return <App ui={useUIProviderProps()} />;
};
export default MyApp;
```

###### In a NextJs App

```tsx
import { App } from "@reactionable/core/lib/app/App";
import { useUIProviderProps } from "@reactionable/ui-bootstrap";
import { ReactElement } from "react";

const MyApp = ({ Component, pageProps }): ReactElement => {
  return (
    <App ui={useUIProviderProps()}>
      <Component {...pageProps} />
    </App>
  );
};
export default MyApp;
```
