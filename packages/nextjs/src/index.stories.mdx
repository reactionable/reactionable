#### Setup App

```tsx
import { App } from "@reactionable/core/lib/app/App";
import { CssInJsWrapper, useRouterProviderProps } from "@reactionable/nextjs";
import { ReactElement } from "react";

const MyApp = ({ Component, pageProps }): ReactElement => {
  return (
    <CssInJsWrapper>
      <App router={useRouterProviderProps()}>
        <Component {...pageProps} />
      </App>
    </CssInJsWrapper>
  );
};
export default MyApp;
```
