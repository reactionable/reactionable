#### Setup App

###### In a React App

```tsx
import { App } from "@reactionable/core/lib/app/App";
import { useUIProviderProps } from "@reactionable/ui-material";
import { ReactElement } from "react";

const MyApp = (): ReactElement => {
  return <App ui={useUIProviderProps()} />;
};
export default MyApp;
```

###### In a NextJs App

```tsx
import { App } from "@reactionable/core/lib/app/App";
import { useUIProviderProps } from "@reactionable/ui-material";
import { ReactElement } from "react";

const MyApp = ({ Component, pageProps }): ReactElement => {
  return (
    <App ui={useUIProviderProps()}>
      <Component {...pageProps} />
    </App>
  );
};
export default MyApp;
```

#### FAQ

##### It is possible to custom UI Context provider props (<LinkTo kind="UI Material/Components/UI">Documentation</LinkTo>)

```tsx
import { ReactElement } from "react";
import { App } from "@reactionable/core/lib/app/App";
import { useUIProviderProps, IUIProviderProps } from "@reactionable/ui-material";

const MyApp = (): ReactElement => {
  const ui: IUIProviderProps = {
    ...useUIProviderProps(),
    theme: {
      palette: {
        primary: { main: "#FF0000" },
      },
    },
  };

  return <App ui={ui} />;
};

export default MyApp;
```
